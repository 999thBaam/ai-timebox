# AI Timebox iOS App — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Flutter iOS app where users brain-dump tasks, get an LLM-generated smart schedule, and the Bayesian belief system learns their patterns over time.

**Architecture:** Local-first Flutter app with SQLite storage, Claude API for scheduling/insights, and a ported Bayesian belief engine. Five screens: daily planner (main), week view, end-of-day review, setup, and settings. Spec lists `scheduler_service.dart`, `task_parser.dart`, `insight_generator.dart` as separate services — this plan consolidates them into `llm_service.dart` (single API layer) and `offline_scheduler.dart` for simplicity.

**Tech Stack:** Flutter/Dart, sqflite, Claude Sonnet API (http), speech_to_text, flutter_secure_storage, flutter_local_notifications

**Spec:** `docs/superpowers/specs/2026-03-26-ios-app-redesign-design.md`

```dart
// lib/models/belief.dart
import 'dart:math';
import 'enums.dart';

const double confidenceCap = 0.95;
const double confidenceFloor = 0.2;
const double confidenceDecayPerDay = 0.01;

class Belief {
  final BeliefParameter parameter;
  final double value;
  final double confidence;
  final DateTime lastUpdated;
  final int evidenceCount;

  Belief({
    required this.parameter,
    required this.value,
    this.confidence = 0.3,
    DateTime? lastUpdated,
    this.evidenceCount = 0,
  }) : lastUpdated = lastUpdated ?? DateTime.now();

  Belief update({required double observation, required double signalWeight}) {
    final adjustedObs = _adjustObservation(observation);
    var newValue = (value * confidence + adjustedObs * signalWeight) /
        (confidence + signalWeight);

    if (parameter == BeliefParameter.peakEnergy) {
      newValue = newValue % 24;
    }

    final newConfidence = min(confidence + signalWeight * 0.1, confidenceCap);

    return Belief(
      parameter: parameter,
      value: newValue,
      confidence: newConfidence,
      lastUpdated: DateTime.now(),
      evidenceCount: evidenceCount + 1,
    );
  }

  Belief withDecay() {
    final daysSince =
        DateTime.now().difference(lastUpdated).inSeconds / 86400.0;
    final decayed = max(confidence - daysSince * confidenceDecayPerDay, confidenceFloor);
    return Belief(
      parameter: parameter,
      value: value,
      confidence: decayed,
      lastUpdated: lastUpdated,
      evidenceCount: evidenceCount,
    );
  }

  double _adjustObservation(double observation) {
    if (parameter != BeliefParameter.peakEnergy) return observation;
    final diff = (value - observation).abs();
    if (diff > 12) {
      return observation < value ? observation + 24 : observation - 24;
    }
    return observation;
  }

  Map<String, dynamic> toMap() => {
    'parameter': parameter.name,
    'value': value,
    'confidence': confidence,
    'last_updated': lastUpdated.toIso8601String(),
    'evidence_count': evidenceCount,
  };

  factory Belief.fromMap(Map<String, dynamic> map) => Belief(
    parameter: BeliefParameter.values.byName(map['parameter'] as String),
    value: (map['value'] as num).toDouble(),
    confidence: (map['confidence'] as num).toDouble(),
    lastUpdated: DateTime.parse(map['last_updated'] as String),
    evidenceCount: map['evidence_count'] as int,
  );
}
```

```dart
// lib/services/belief_engine.dart
import 'dart:math';
import '../models/belief.dart';
import '../models/enums.dart';

class BeliefEngine {
  static const _defaults = <BeliefParameter, double>{
    BeliefParameter.peakEnergy: 10.0,
    BeliefParameter.deepWorkTolerance: 60.0,
    BeliefParameter.contextSwitchCost: 0.5,
    BeliefParameter.chaosTolerance: 0.4,
    BeliefParameter.meetingTolerance: 0.5,
    BeliefParameter.recoveryRate: 0.10,
  };

  static const _signalWeights = (
    blockCompletion: 0.3,
    reschedule: 0.4,
    selfScheduled: 0.2,
    energySelfReport: 0.5,
  );

  Map<BeliefParameter, Belief> getDefaultBeliefs() {
    return {
      for (final param in BeliefParameter.values)
        param: Belief(parameter: param, value: _defaults[param]!, confidence: 0.3),
    };
  }

  Map<BeliefParameter, Belief> processTaskCompletion(
    Map<BeliefParameter, Belief> beliefs, {
    required double taskStartHour,
    required int durationMinutes,
    double? minutesSinceLastBlock,
  }) {
    final result = Map<BeliefParameter, Belief>.from(beliefs);
    const w = 0.3; // blockCompletion weight

    result[BeliefParameter.peakEnergy] =
        beliefs[BeliefParameter.peakEnergy]!.update(observation: taskStartHour, signalWeight: w);
    result[BeliefParameter.deepWorkTolerance] =
        beliefs[BeliefParameter.deepWorkTolerance]!.update(observation: durationMinutes.toDouble(), signalWeight: w);

    if (minutesSinceLastBlock != null && minutesSinceLastBlock < 15) {
      final current = beliefs[BeliefParameter.contextSwitchCost]!.value;
      result[BeliefParameter.contextSwitchCost] =
          beliefs[BeliefParameter.contextSwitchCost]!.update(observation: max(current - 0.15, 0.0), signalWeight: w);
    }
    return result;
  }

  Map<BeliefParameter, Belief> processSkip(
    Map<BeliefParameter, Belief> beliefs, {
    required double skippedHour,
    bool isMeetingAdjacent = false,
  }) {
    final result = Map<BeliefParameter, Belief>.from(beliefs);
    const w = 0.3;
    final opposite = skippedHour > 12 ? skippedHour - 12 : skippedHour + 12;

    result[BeliefParameter.peakEnergy] =
        beliefs[BeliefParameter.peakEnergy]!.update(observation: opposite, signalWeight: w);

    if (isMeetingAdjacent) {
      final current = beliefs[BeliefParameter.meetingTolerance]!.value;
      result[BeliefParameter.meetingTolerance] =
          beliefs[BeliefParameter.meetingTolerance]!.update(observation: max(current - 0.15, 0.0), signalWeight: w);
    }
    return result;
  }

  Map<BeliefParameter, Belief> processReschedule(
    Map<BeliefParameter, Belief> beliefs, {
    required double newHour,
  }) {
    final result = Map<BeliefParameter, Belief>.from(beliefs);
    result[BeliefParameter.peakEnergy] =
        beliefs[BeliefParameter.peakEnergy]!.update(observation: newHour, signalWeight: 0.4);
    return result;
  }

  Map<BeliefParameter, Belief> processEnergyReport(
    Map<BeliefParameter, Belief> beliefs, {
    required double currentHour,
    required String level,
  }) {
    const energyMap = {'low': 0.3, 'ok': 0.6, 'great': 0.9};
    final energyValue = energyMap[level] ?? 0.5;
    if (energyValue < 0.7) return beliefs;

    final result = Map<BeliefParameter, Belief>.from(beliefs);
    result[BeliefParameter.peakEnergy] =
        beliefs[BeliefParameter.peakEnergy]!.update(observation: currentHour, signalWeight: 0.5);
    return result;
  }

  Map<BeliefParameter, Belief> decayAll(Map<BeliefParameter, Belief> beliefs) {
    return {for (final e in beliefs.entries) e.key: e.value.withDecay()};
  }
}
```

```dart
enum HealthPhase { silent, gentleNudge, patternForming, habitBuilding, adapted }

class HealthTracker {
  final int appDayCount;
  final int acceptedNudgesThisWeek;
  final bool skippedFullWeek;

  HealthTracker({
    required this.appDayCount,
    required this.acceptedNudgesThisWeek,
    required this.skippedFullWeek,
  });

  HealthPhase get currentPhase {
    if (skippedFullWeek) return HealthPhase.gentleNudge;
    if (appDayCount <= 3) return HealthPhase.silent;
    if (appDayCount <= 7) return HealthPhase.gentleNudge;
    if (appDayCount <= 21) {

    }
    if (appDayCount <= 60) return HealthPhase.habitBuilding;
    return HealthPhase.adapted;
  }
```

```dart
// lib/services/offline_scheduler.dart
import '../models/task.dart';
import '../models/belief.dart';
import '../models/schedule_slot.dart';
import '../models/enums.dart';

class OfflineScheduler {
  static List<ScheduleSlot> generate({
    required List<Task> tasks,
    required Map<BeliefParameter, Belief> beliefs,
    required String wakeTime,
    required String sleepTime,
  }) {
    final peakHour = beliefs[BeliefParameter.peakEnergy]?.value ?? 10.0;
    final switchCost = beliefs[BeliefParameter.contextSwitchCost]?.value ?? 0.5;
    final bufferMinutes = switchCost > 0.6 ? 15 : (switchCost > 0.3 ? 10 : 5);

    // Sort: deep_focus first, then quick, then outing
    final sorted = List<Task>.from(tasks)..sort((a, b) {
      const order = {TaskType.deepFocus: 0, TaskType.call: 1, TaskType.quick: 1, TaskType.outing: 2};
      return (order[a.taskType] ?? 1).compareTo(order[b.taskType] ?? 1);
    });

    final slots = <ScheduleSlot>[];
    var currentHour = peakHour.floor();
    var currentMinute = 0;

    for (final task in sorted) {
      // Place task
      final startTime = '${currentHour.toString().padLeft(2, '0')}:${currentMinute.toString().padLeft(2, '0')}';
      final endMinutes = currentHour * 60 + currentMinute + task.estimatedMinutes;
      final endHour = endMinutes ~/ 60;
      final endMin = endMinutes % 60;
      final endTime = '${endHour.toString().padLeft(2, '0')}:${endMin.toString().padLeft(2, '0')}';

      slots.add(ScheduleSlot(
        taskId: task.id,
        slotType: SlotType.task,
        startTime: startTime,
        endTime: endTime,
        title: task.title,
        pillar: task.pillar,
      ));

      // Add buffer
      final bufferEnd = endMinutes + bufferMinutes;
      slots.add(ScheduleSlot(
        slotType: SlotType.buffer,
        startTime: endTime,
        endTime: '${(bufferEnd ~/ 60).toString().padLeft(2, '0')}:${(bufferEnd % 60).toString().padLeft(2, '0')}',
        title: 'Buffer',
      ));

      currentHour = bufferEnd ~/ 60;
      currentMinute = bufferEnd % 60;
    }
    return slots;
  }
}
```

```dart
  Future<Map<String, dynamic>?> _callClaude(String systemPrompt, String userMessage) async {
    final apiKey = await getApiKey();
    if (apiKey == null || apiKey.isEmpty) return null;

    // Retry once on malformed JSON (spec requirement)
    for (var attempt = 0; attempt < 2; attempt++) {
      try {
        final response = await http.post(
          Uri.parse(_apiUrl),
          headers: {
            'Content-Type': 'application/json',
            'x-api-key': apiKey,
            'anthropic-version': '2023-06-01',
          },
          body: jsonEncode({
            'model': 'claude-sonnet-4-20250514',
            'max_tokens': 2048,
            'system': systemPrompt,
            'messages': [{'role': 'user', 'content': userMessage}],
          }),
        );

        if (response.statusCode != 200) return null;

        final body = jsonDecode(response.body);
        final text = body['content'][0]['text'] as String;
        // Try parsing as Map first, then as List wrapped in Map
        final parsed = jsonDecode(text);
        if (parsed is Map<String, dynamic>) return parsed;
        if (parsed is List) return {'items': parsed};
        return null;
      } on FormatException {
        if (attempt == 1) return null; // Second attempt failed, give up
        continue; // Retry once
      } catch (e) {
        return null; // Network error — no retry, use offline fallback
      }
    }
    return null;

    final undoneList = undoneTasks.map((t) =>
```

```dart
// lib/services/speech_service.dart
import 'package:speech_to_text/speech_to_text.dart';

class SpeechService {
  final SpeechToText _speech = SpeechToText();
  bool _isAvailable = false;

  Future<bool> init() async {
    _isAvailable = await _speech.initialize();
    return _isAvailable;
  }

  bool get isAvailable => _isAvailable;

  Future<void> startListening({required Function(String) onResult}) async {
    if (!_isAvailable) return;
    await _speech.listen(
      onResult: (result) {
        if (result.finalResult) {
          onResult(result.recognizedWords);
        }
      },
      listenFor: const Duration(seconds: 30),
      pauseFor: const Duration(seconds: 3),
    );
  }

  Future<void> stopListening() async => _speech.stop();
}
```
